isSameWeek

给定日期是否在同一周 类型 说明 Boolean 是否在同一周 
            // Are 31 August 2014 and 4 September 2014 in the same week?
            var result = isSameWeek(
              new Date(2014, 7, 31),
              new Date(2014, 8, 4)
            )
            //=> true
          
          // If week starts with Monday,
          // are 31 August 2014 and 4 September 2014 in the same week?
          var result = isSameWeek(
            new Date(2014, 7, 31),
            new Date(2014, 8, 4),
            ｛weekStartsOn: 1｝
          )
          //=> false
