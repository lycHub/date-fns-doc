distanceInWordsToNow

以语言的形式返回给定日期与当前时间的间隔 Distance between dates Result {{data.distance}} {{data.result}} 设置选项：includeSeconds == true Distance between dates Result {{data.distance}} {{data.result}}
类型 说明 String 日期间隔的描述。 
          // 如果今天是2015-1-1，那么2014-7-2距今多久?
          var result = distanceInWordsToNow(
            new Date(2014, 6, 2)
          )
          //=> '6 months'
          
          // 如果今天是2015-1-1  00:00:00，那么2015-1-1 00:00:15距今多久？  包括秒
          var result = distanceInWordsToNow(
            new Date(2015, 0, 1, 0, 0, 15),
            ｛includeSeconds: true｝
          )
          //=> 'less than 20 seconds'
        
         // 如果今天是2015-1-1，那么2016-1-1距今多久？  带后缀
          var result = distanceInWordsToNow(
            new Date(2016, 0, 1),
            ｛addSuffix: true｝
          )
          //=> 'in about 1 year'
        
          // 如果今天是2015-1-1，那么2016-8-1距今多久？  国际化
          var eoLocale = require('date-fns/locale/eo')
          var result = distanceInWordsToNow(
