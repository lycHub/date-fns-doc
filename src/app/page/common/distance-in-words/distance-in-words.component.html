distanceInWords

以语言的形式返回给定日期的间隔 Distance between dates Result {{data.distance}} {{data.result}} 设置选项：includeSeconds == true Distance between dates Result {{data.distance}} {{data.result}}
类型 说明 String 给定日期的间隔描述。 
            // 2014-7-2与2015-1-1相隔多久？
            const result = distanceInWords(
              new Date(2014, 6, 2),
              new Date(2015, 0, 1)
            )
            //=> '6 months'
          
           // 2015年1月1日00:00:15到2015年1月1日00:00:00间隔多少？包括秒
          const result = distanceInWords(
            new Date(2015, 0, 1, 0, 0, 15),
            new Date(2015, 0, 1, 0, 0, 0),
            ｛includeSeconds: true｝
          )
          //=> 'less than 20 seconds'
          
          // 2015年1月1日到2016年1月1日间隔多少？带后缀
          const result = distanceInWords(
            new Date(2016, 0, 1),
            new Date(2015, 0, 1),
            ｛addSuffix: true｝
          )
          //=> 'about 1 year ago'
        
          // 2016年8月1日至2015年1月1日间隔多少？ 国际化
          const eoLocale = require('date-fns/locale/eo')
          const result = distanceInWords(
            new Date(2016, 7, 1),
            new Date(2015, 0, 1),
            ｛locale: eoLocale｝
          )
          //=> 'pli ol 1 jaro'
