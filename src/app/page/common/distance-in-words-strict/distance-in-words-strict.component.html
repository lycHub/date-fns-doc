distanceInWordsStrict

用更严谨的单位语言返回给定日期之间的距离。不使用像‘几乎’，‘结束’，‘小于’之类的形容词。 Distance between dates Result {{data.distance}} {{data.result}}
类型 说明 String 给定日期的间隔描述。 
            // 2014-7-2与2015-1-1相隔多久？
            var result = distanceInWordsStrict(
              new Date(2014, 6, 2),
              new Date(2015, 0, 1)
            )
            //=> '6 months'
          
           // 2015年1月1日00:00:15到2015年1月1日00:00:00间隔多少？包括秒
          var result = distanceInWordsStrict(
            new Date(2015, 0, 1, 0, 0, 15),
            new Date(2015, 0, 1, 0, 0, 0),
          )
          //=> '15 seconds'
          
          // 2015年1月1日到2016年1月1日间隔多少？带后缀
          const result = distanceInWordsStrict(
            new Date(2016, 0, 1),
            new Date(2015, 0, 1),
            ｛addSuffix: true｝
          )
          //=> '1 year ago'
        
           // 2015年1月1日到2016年1月1日间隔多少？单位：minutes
          var result = distanceInWordsStrict(
            new Date(2016, 0, 1),
            new Date(2015, 0, 1),
            ｛unit: 'm'｝
          )
          //=> '525600 minutes'
        
         // 2015年1月1日到2015年1月28日间隔多少？  四舍五入
        var result = distanceInWordsStrict(
          new Date(2015, 0, 28),
          new Date(2015, 0, 1),
          ｛unit: 'M', partialMethod: 'ceil'｝
        )
        //=> '1 month'
        
          // 2016年8月1日至2015年1月1日间隔多少？ 国际化
          const eoLocale = require('date-fns/locale/eo')
          const result = distanceInWordsStrict(
            new Date(2016, 7, 1),
            new Date(2015, 0, 1),
            ｛locale: eoLocale｝
          )
          //=> '1 jaro'
